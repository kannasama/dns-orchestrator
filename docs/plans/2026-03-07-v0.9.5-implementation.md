```cpp
CROW_ROUTE(app, "/api/v1/providers/health").methods("GET"_method)(
    [this](const crow::request& req) -> crow::response {
      try {
        auto rcCtx = authenticate(_amMiddleware, req);
        requireRole(rcCtx, "viewer");

        auto vProviders = _prRepo.listAll();
        nlohmann::json jResults = nlohmann::json::array();

        for (const auto& pr : vProviders) {
          std::string sStatus = "error";
          std::string sMessage;
          try {
            auto upProvider = dns::providers::ProviderFactory::create(
                pr.sType, pr.sApiEndpoint, _csService.decrypt(pr.sEncryptedToken), pr.jConfig);
            auto hs = upProvider->testConnectivity();
            if (hs == common::HealthStatus::Ok) {
              sStatus = "healthy";
              sMessage = "Connected successfully";
            } else if (hs == common::HealthStatus::Degraded) {
              sStatus = "unreachable";
              sMessage = "Provider responded but degraded";
            } else {
              sStatus = "unreachable";
              sMessage = "Could not reach provider";
            }
          } catch (const std::exception& e) {
            sStatus = "error";
            sMessage = e.what();
          }

          jResults.push_back({
              {"id", pr.iId},
              {"name", pr.sName},
              {"type", pr.sType},
              {"status", sStatus},
              {"message", sMessage},
          });
        }
        return jsonResponse(200, jResults);
      } catch (const common::AppError& e) {
        return errorResponse(e);
      }
    });
```

```cpp
#include "dal/GroupRepository.hpp"
```

```cpp
namespace dns::dal {

GroupRepository::GroupRepository(ConnectionPool& cpPool) : _cpPool(cpPool) {}

std::vector<GroupRow> GroupRepository::listAll() {
  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);
  auto result = txn.exec(
      "SELECT g.id, g.name, g.role, g.description, "
      "  EXTRACT(EPOCH FROM g.created_at)::bigint, "
      "  COUNT(gm.user_id)::int "
      "FROM groups g "
      "LEFT JOIN group_members gm ON gm.group_id = g.id "
      "GROUP BY g.id ORDER BY g.name");
  txn.commit();

  std::vector<GroupRow> vGroups;
  for (const auto& row : result) {
    vGroups.push_back({
        row[0].as<int64_t>(),
        row[1].as<std::string>(),
        row[2].as<std::string>(),
        row[3].as<std::string>(""),
        row[5].as<int>(),
        std::chrono::system_clock::time_point(std::chrono::seconds(row[4].as<int64_t>())),
    });
  }
  return vGroups;
}

std::optional<GroupRow> GroupRepository::findById(int64_t iGroupId) {
  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);
  auto result = txn.exec(
      "SELECT g.id, g.name, g.role, g.description, "
      "  EXTRACT(EPOCH FROM g.created_at)::bigint, "
      "  COUNT(gm.user_id)::int "
      "FROM groups g "
      "LEFT JOIN group_members gm ON gm.group_id = g.id "
      "WHERE g.id = $1 GROUP BY g.id",
      pqxx::params{iGroupId});
  txn.commit();

  if (result.empty()) return std::nullopt;
  const auto& row = result[0];
  return GroupRow{
      row[0].as<int64_t>(),
      row[1].as<std::string>(),
      row[2].as<std::string>(),
      row[3].as<std::string>(""),
      row[5].as<int>(),
      std::chrono::system_clock::time_point(std::chrono::seconds(row[4].as<int64_t>())),
  };
}

int64_t GroupRepository::create(const std::string& sName, const std::string& sRole,
                                 const std::string& sDescription) {
  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);
  try {
    auto result = txn.exec(
        "INSERT INTO groups (name, role, description) VALUES ($1, $2, $3) RETURNING id",
        pqxx::params{sName, sRole, sDescription});
    txn.commit();
    return result[0][0].as<int64_t>();
  } catch (const pqxx::unique_violation&) {
    throw common::ConflictError("group_exists", "Group name already exists");
  }
}

void GroupRepository::update(int64_t iGroupId, const std::string& sName,
                              const std::string& sRole, const std::string& sDescription) {
  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);
  auto result = txn.exec(
      "UPDATE groups SET name = $1, role = $2, description = $3 WHERE id = $4",
      pqxx::params{sName, sRole, sDescription, iGroupId});
  txn.commit();
  if (result.affected_rows() == 0)
    throw common::NotFoundError("group_not_found", "Group not found");
}

void GroupRepository::deleteGroup(int64_t iGroupId) {
  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);

  // Check if this is any user's only group
  auto check = txn.exec(
      "SELECT gm.user_id FROM group_members gm "
      "WHERE gm.group_id = $1 "
      "AND NOT EXISTS ("
      "  SELECT 1 FROM group_members gm2 "
      "  WHERE gm2.user_id = gm.user_id AND gm2.group_id != $1"
      ")",
      pqxx::params{iGroupId});

  if (!check.empty())
    throw common::ConflictError("group_sole_membership",
        "Cannot delete: group is the only group for one or more users");

  txn.exec("DELETE FROM groups WHERE id = $1", pqxx::params{iGroupId});
  txn.commit();
}

std::vector<std::pair<int64_t, std::string>> GroupRepository::listMembers(int64_t iGroupId) {
  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);
  auto result = txn.exec(
      "SELECT u.id, u.username FROM users u "
      "JOIN group_members gm ON gm.user_id = u.id "
      "WHERE gm.group_id = $1 ORDER BY u.username",
      pqxx::params{iGroupId});
  txn.commit();

  std::vector<std::pair<int64_t, std::string>> vMembers;
  for (const auto& row : result)
    vMembers.emplace_back(row[0].as<int64_t>(), row[1].as<std::string>());
  return vMembers;
}
```

```cpp
void UserRoutes::registerRoutes(crow::SimpleApp& app) {
  // GET /api/v1/users
  CROW_ROUTE(app, "/api/v1/users").methods("GET"_method)(
      [this](const crow::request& req) -> crow::response {
        try {
          auto rcCtx = authenticate(_amMiddleware, req);
          requireRole(rcCtx, "admin");
          auto vUsers = _urRepo.listAll();
          nlohmann::json jUsers = nlohmann::json::array();
          for (const auto& u : vUsers) {
            auto vGroups = _urRepo.listGroupsForUser(u.iId);
            nlohmann::json jGroups = nlohmann::json::array();
            for (const auto& [iGid, sGname] : vGroups)
              jGroups.push_back({{"id", iGid}, {"name", sGname}});
            jUsers.push_back({
                {"id", u.iId}, {"username", u.sUsername}, {"email", u.sEmail},
                {"auth_method", u.sAuthMethod}, {"is_active", u.bIsActive},
                {"groups", jGroups},
            });
          }
          return jsonResponse(200, jUsers);
        } catch (const common::AppError& e) { return errorResponse(e); }
      });

  // POST /api/v1/users
  CROW_ROUTE(app, "/api/v1/users").methods("POST"_method)(
      [this](const crow::request& req) -> crow::response {
        try {
          auto rcCtx = authenticate(_amMiddleware, req);
          requireRole(rcCtx, "admin");
          auto jBody = nlohmann::json::parse(req.body);

          auto sUsername = jBody.at("username").get<std::string>();
          auto sEmail = jBody.at("email").get<std::string>();
          auto sPassword = jBody.at("password").get<std::string>();
          auto bForceChange = jBody.value("force_password_change", true);
          auto jGroupIds = jBody.value("group_ids", nlohmann::json::array());

          RequestValidator::validateUsername(sUsername);
          RequestValidator::validatePassword(sPassword);

          auto sHash = dns::security::CryptoService::hashPassword(sPassword);
          auto iUserId = _urRepo.create(sUsername, sEmail, sHash);

          if (bForceChange)
            _urRepo.setForcePasswordChange(iUserId, true);

          for (const auto& jGid : jGroupIds)
            _urRepo.addToGroup(iUserId, jGid.get<int64_t>());

          return jsonResponse(201, {{"id", iUserId}});
        } catch (const nlohmann::json::exception&) { return invalidJsonResponse(); }
          catch (const common::AppError& e) { return errorResponse(e); }
      });

              {"auth_method", oUser->sAuthMethod}, {"is_active", oUser->bIsActive},
              {"force_password_change", oUser->bForcePasswordChange}, {"groups", jGroups},

      });
```

```cpp
// POST /api/v1/auth/change-password
CROW_ROUTE(app, "/api/v1/auth/change-password").methods("POST"_method)(
    [this](const crow::request& req) -> crow::response {
      try {
        auto rcCtx = authenticate(_amMiddleware, req);
        auto jBody = nlohmann::json::parse(req.body);
        auto sCurrentPassword = jBody.at("current_password").get<std::string>();
        auto sNewPassword = jBody.at("new_password").get<std::string>();

        // Verify current password
        auto oUser = _urRepo.findById(rcCtx.iUserId);
        if (!oUser) throw common::AuthenticationError("user_not_found", "User not found");
        if (!CryptoService::verifyPassword(sCurrentPassword, oUser->sPasswordHash))
          throw common::AuthenticationError("invalid_password", "Current password is incorrect");

        RequestValidator::validatePassword(sNewPassword);
        auto sHash = CryptoService::hashPassword(sNewPassword);
        _urRepo.updatePassword(rcCtx.iUserId, sHash);
        _urRepo.setForcePasswordChange(rcCtx.iUserId, false);
        return jsonResponse(200, {{"message", "Password changed"}});
      } catch (const nlohmann::json::exception&) { return invalidJsonResponse(); }
        catch (const common::AppError& e) { return errorResponse(e); }
    });
```

```typescript
function relativeTime(iso: string | null): string {
```
